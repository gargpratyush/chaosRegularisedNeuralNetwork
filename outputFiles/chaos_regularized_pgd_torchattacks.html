import torch
import torch.nn as nn
import torch.optim as optim
import torchvision
import torchvision.transforms as transforms
import torch.backends.cudnn as cudnn
from torchvision.models import resnet18
import torchattacks
from resnet import *

import time

start_time = time.time()

# Set up device
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# Load ResNet18 from file
net = ResNet18()
net = net.to(device)
if device == 'cuda':
    net = torch.nn.DataParallel(net)
    cudnn.benchmark = True

# Load CIFAR-10 dataset
transform = transforms.Compose([
    transforms.RandomCrop(32, padding=4),
    transforms.RandomHorizontalFlip(),
    transforms.ToTensor(),
    transforms.Normalize((0.4914, 0.4822, 0.4465), (0.2023, 0.1994, 0.2010))
])

trainset = torchvision.datasets.CIFAR10(root='./data', train=True, download=True, transform=transform)
trainloader = torch.utils.data.DataLoader(trainset, batch_size=128, shuffle=True, num_workers=2)

testset = torchvision.datasets.CIFAR10(root='./data', train=False, download=True, transform=transform)
testloader = torch.utils.data.DataLoader(testset, batch_size=100, shuffle=False, num_workers=2)

# Define the ResNet-18 Model
class ResNet18WithOutput(nn.Module):
    def __init__(self):
        super(ResNet18WithOutput, self).__init__()
        self.resnet18 = net

    def forward(self, x):
        output = self.resnet18(x)
        return output

model = ResNet18WithOutput().to(device)

def adjust_learning_rate(optimizer, epoch):
    lr = 0.01
    if epoch >= 100:
        lr /= 10
    if epoch >= 150:
        lr /= 10
    for param_group in optimizer.param_groups:
        param_group['lr'] = lr

# Custom Chaos Regularized Loss with Adversarial Perturbations using torchattacks.PGD
class ChaosRegularizedLoss(nn.Module):
    def __init__(self, model, lambda_chaos=0.8, eps=8/255, alpha=3/255, iters=10):
        super(ChaosRegularizedLoss, self).__init__()
        self.lambda_chaos = lambda_chaos
        self.cross_entropy_loss = nn.CrossEntropyLoss()
        self.pgd_attack = torchattacks.PGD(model, eps=eps, alpha=alpha, steps=iters)

    def forward(self, model, inputs, targets):
        # Compute standard task-specific loss using cross-entropy
        outputs = model(inputs)
        task_loss = self.cross_entropy_loss(outputs, targets)

        adv_inputs = self.pgd_attack(inputs, targets)
        outputs_adv = model(adv_inputs)

        # Chaos regularization term (mean squared difference divided by the perturbation)
        perturbation = adv_inputs - inputs
        chaos_loss = torch.mean((outputs - outputs_adv).pow(2) / perturbation.pow(2).mean())

        # Combined loss
        total_loss = task_loss + self.lambda_chaos * chaos_loss
        return total_loss

criterion = ChaosRegularizedLoss(model, lambda_chaos=0.8, eps=8/255, alpha=3/255, iters=10)
optimizer = optim.Adam(model.parameters(), lr=0.01)

criterion_benign = nn.CrossEntropyLoss()
optimizer_benign = optim.SGD(net.parameters(), lr=0.01, momentum=0.9, weight_decay=0.0002)

# Train the Model

test_adv_loss_list = []
test_benign_loss_list = []

test_adv_accuracy = []
test_benign_accuracy = []

# Define the PGD attack
epsilon = 8/255  # Perturbation
alpha = 2/255    # Step size
steps = 20        # Number of iterations

pgdattack = torchattacks.PGD(model, eps=epsilon, alpha=alpha, steps=steps)

for epoch in range(200): 
    adjust_learning_rate(optimizer, epoch)
    model.train()
    running_loss = 0.0
    
    for i, data in enumerate(trainloader, 0):
        inputs, targets = data
        inputs, targets = inputs.to(device), targets.to(device)
        
        optimizer.zero_grad()
        
        loss = criterion(model, inputs, targets)
        loss.backward()
        optimizer.step()
        
        running_loss += loss.item()
        if i % 100 == 99:  # Print every 100 mini-batches
            print(f'[Epoch {epoch + 1}, Batch {i + 1}] loss: {running_loss / 100:.3f}')
            # running_loss = 0.0
        
    training_loss.append(running_loss / 100)
    
    

    # Function to test the model on adversarial examples
    adv_correct = 0
    total = 0
    model.eval()
    test_adv_loss = 0.0
    
    for data in testloader:
        images, labels = data
        images, labels = images.to(device), labels.to(device)
        adv_images = pgdattack(images, labels) # Generate adversarial examples
        outputs = model(adv_images) # Forward pass the adversarial examples through the model
        _, predicted = torch.max(outputs.data, 1)
        total += labels.size(0)
        adv_correct += (predicted == labels).sum().item()
        loss = criterion(model, adv_images, labels)
        test_adv_loss += loss.item()

    test_adv_loss_list.append(test_adv_loss / len(testloader))
    print(f'Accuracy of the model on adversarial examples: {100 * adv_correct / total:.2f}%')
    test_adv_accuracy.append(100 * adv_correct / total)
    
    benign_loss = 0
    benign_correct = 0
    total = 0
    
    for batch_idx, (inputs, targets) in enumerate(testloader):
        inputs, targets = inputs.to(device), targets.to(device)
        outputs = model(inputs)
        _, predicted = torch.max(outputs.data, 1)
        total += targets.size(0)
        benign_correct += (predicted == targets).sum().item()
        loss = criterion(model, images, labels)
        benign_loss += loss.item()
        
    test_benign_loss_list.append(benign_loss / len(testloader))
    print(f'[Epoch {epoch + 1}] test benign loss: {benign_loss / len(testloader):.3f}')
    test_benign_accuracy.append(100 * benign_correct / total)
